Javascript Loading Test - #5

Loading Javascript file on daemond by using document.createElement() -- The orginal article is Javascript Async Loading

The Javascript file is alert.js

Click to load alert.js

The test source code as below: 

<script language="javascript" type="text/javascript">
    function loadjs(script_filename) {
        var script = document.createElement('script');
        script.setAttribute('type', 'text/javascript');
        script.setAttribute('src', script_filename);
        script.setAttribute('id', 'coolshell_script_id');

        script_id = document.getElementById('coolshell_script_id');
        if(script_id){
            document.getElementsByTagName('head')[0].removeChild(script_id);
        }
        document.getElementsByTagName('head')[0].appendChild(script);
    }

    function LoadJS(){
        var script = './alert.js';
        loadjs(script);
    }

</script>

<p style="cursor: pointer" onclick="LoadJS()">Click to load alert.js </p>
